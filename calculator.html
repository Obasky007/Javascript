// CALCULATOR
function add() {
  let firstValue = document.getElementById('first').value;
  let secondValue = document.getElementById('second').value;
  let resultElement = document.getElementById('result');
  if (firstValue === "" || secondValue === "") {
    resultElement.textContent = "Please enter a number in both fields";
    resultElement.style.color = "red";
    return;
  }
  resultElement.style.color = "green";
  document.getElementById('p').textContent = "+";
  let result = Number(firstValue) + Number(secondValue);
  resultElement.textContent = `The sum of ${firstValue} and ${secondValue} is ${result}`;
}

// RETIREMENT CALCULATOR
function retirement() {
  let age = document.getElementById('age').value;
  let retired = 60 - age;
  let supposed = age - 60;
  let output = document.getElementById('retire');

  if (age === '') {
    output.textContent = 'Type In Your Age';
    output.style.color = 'red';
  } else if (age == 60) {
    output.style.color = 'orange';
    output.textContent = `You are ${age} years old, you are to retire this year.`;
  } else if (age >= 61) {
    output.style.color = 'blue';
    output.textContent = `You are ${age} years old, you were supposed to retire ${supposed} years ago.`;
  } else {
    output.style.color = 'green';
    output.textContent = `You are ${age} years old, you have ${retired} more years to retire.`;
  }
}
